        const clapButton = document.getElementById("clapButton");

        clapButton.addEventListener("click", function() {
            this.style.backgroundColor = "#f4b6c2"; // 点击后按钮背景变粉色
            this.style.transform = "scale(1.1)"; // 轻微放大效果
            
            // 控制 emoji 数量，随机 3~5 个
            const emojiCount = Math.floor(Math.random() * 1) + 1; 
            const emojiPosition = this.getBoundingClientRect();

            for (let i = 0; i < emojiCount; i++) {
                const newEmoji = document.createElement("span");
                newEmoji.textContent = "💖";
                newEmoji.style.position = "absolute";
                newEmoji.style.left = `${emojiPosition.left + (Math.random() * 30 - 15)}px`; 
                newEmoji.style.top = `${emojiPosition.top}px`; 
                newEmoji.style.fontSize = "24px";
                newEmoji.style.opacity = "1";
                newEmoji.style.animation = "emoji-float 1.5s forwards ease-out";

                document.body.appendChild(newEmoji);

                // 动画结束后删除 emoji
                newEmoji.addEventListener("animationend", () => {
                    newEmoji.remove();
                });
            }

            // 300ms 后恢复按钮原样
            setTimeout(() => {
                this.style.backgroundColor = "#fff";
                this.style.transform = "scale(1)";
            }, 300);
        });

        // 追加 CSS 动画
        const style = document.createElement('style');
        style.innerHTML = `
            @keyframes emoji-float {
                0% {
                    opacity: 1;
                    transform: translateY(0) scale(1);
                }
                100% {
                    opacity: 0;
                    transform: translateY(-120px) scale(0.6) translateX(${Math.random() * 30 - 15}px);
                }
            }
        `;
        document.head.appendChild(style);
    </script>  


    <!-- 拍手按钮 -->
    <button id="clapButton" style="
        position: fixed; 
        bottom: 10px; 
        right: 10px; 
        background-color: #fff; 
        border: none; 
        padding: 10px 10px; 
        font-size: 20px; 
        border-radius: 30px; 
        box-shadow: 0 4px 6px rgba(0, 0, 0, 0.1); 
        cursor: pointer; 
        transition: background-color 0.3s, transform 0.2s;
    ">
        💖
    </button>

    <script>
        const clapButton = document.getElementById("clapButton");

        // 生成飘动的 emoji
        function createFloatingEmoji(x, y) {
            const emoji = document.createElement("span");
            emoji.textContent = "💖";
            emoji.style.position = "absolute";
            emoji.style.left = `${x}px`;
            emoji.style.top = `${y}px`;
            emoji.style.fontSize = "24px";
            emoji.style.opacity = "1";
            emoji.style.pointerEvents = "none";
            emoji.style.animation = "emoji-float 1.5s forwards ease-out";
            document.body.appendChild(emoji);
            emoji.addEventListener("animationend", () => emoji.remove());
        }

        // 按钮点击事件
        clapButton.addEventListener("click", function() {
            this.style.backgroundColor = "#f4b6c2";
            this.style.transform = "scale(1.1)";
            const rect = this.getBoundingClientRect();
            createFloatingEmoji(rect.left + rect.width / 2, rect.top);
            setTimeout(() => {
                this.style.backgroundColor = "#fff";
                this.style.transform = "scale(1)";
            }, 300);
        });

        // 监听页面点击事件
        document.addEventListener("click", function(event) {
            if (!clapButton.contains(event.target)) {
                createFloatingEmoji(event.clientX, event.clientY);
            }
        });

        // CSS 动画
        const style = document.createElement('style');
        style.innerHTML = `
            @keyframes emoji-float {
                0% {
                    opacity: 1;
                    transform: translateY(0) scale(1);
                }
                100% {
                    opacity: 0;
                    transform: translateY(-120px) scale(0.6) translateX(${Math.random() * 30 - 15}px);
                }
            }
        `;
        document.head.appendChild(style);
    </script>
